#include <bits/stdc++.h>
#define maxn 1005
using namespace std;

typedef pair<int,int> II;
II c[maxn];
int n;
int a[maxn],b[maxn];
int dd[maxn];

bool giao(int i,int j) {
    if(b[i]>=a[j]) return true;
    return false;
}

void ghep(int i,int j) {
    b[i]=b[j];
    dd[j]=1;
}

int main() {
    freopen("a.inp","r",stdin);
    n=0;
    int x,y;
    while(cin>>x>>y) {
        c[++n]=II(x,y);
    }
    sort(c+1,c+n+1);
    for(int i=1;i<=n;++i) a[i]=c[i].first,b[i]=c[i].second;
    for(int i=1;i<=n;++i) if(dd[i]==0) {
        for(int j=i+1;j<=n;++j) if(giao(i,j)) ghep(i,j);
    }
    for(int i=1;i<=n;++i) if(dd[i]==0) cout<<a[i]<<" "<<b[i]<<"\n";
}
